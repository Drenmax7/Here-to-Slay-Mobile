package org.example;

import java.io.IOException;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.io.OutputStreamWriter;
import java.net.ServerSocket;
import java.net.Socket;
import java.nio.charset.StandardCharsets;
import java.util.HashMap;
import java.util.UUID;
import java.util.concurrent.locks.ReentrantLock;

import org.json.JSONException;
import org.json.JSONObject;

/**
 * Class that permit connexion with clients.
 * Its purpose is to send the game state to clients, so they can update their game view
 * It also gets the user inputs to make a game turn
 *
 */
public class Server {
    //Attributes

    /**
     * Run until close is set to true
     */
    private boolean close;

    /**
     * The port that is listened by the server
     */
    private int port;

    /**
     * Socket on which incoming connexion are put and that permits the server to accept them
     */
    private ServerSocket serverSocket;

    /**
     * List of the client's socket currently connected
     */
    private final HashMap<String,Socket> connectedClient;

    /**
     * List of the client's output stream currently connected onto the server
     */
    private final HashMap<String,ObjectOutputStream> clientOutput;

    /**
     * Lock used to prevent multiple thread to delete sockets from the socket's list at the same time
     */
    private final ReentrantLock lock;

    private String controllerUUID;

    //Getter

    public int getPort() {
        return port;
    }

    //Constructor

    /**
     * Create a new server, listening on the given port
     *
     */
    public Server() {
        connectedClient = new HashMap<>();
        clientOutput = new HashMap<>();
        lock = new ReentrantLock();

//        initializeRegistrationListener();
//        registerService(context);
    }

    //Methods

    /**
     * Start the server, that consist of an infinite loop of accepting incoming connexion
     * Should be run in a thread
     */
    public void run() {

        // Initialize a server socket on the next available port.
        try(ServerSocket serverSocket = new ServerSocket(6666)){
            // Store the chosen port.
            port = serverSocket.getLocalPort();
            System.out.println("server is ready to run on port " + port);

            this.serverSocket = serverSocket;
            close = false;

            // As long as the server is open, it accepts connections
            while (!close) {
                try {

                    Socket socket = serverSocket.accept();
                    ObjectInputStream input = new ObjectInputStream(socket.getInputStream());
                    ObjectOutputStream output = new ObjectOutputStream(socket.getOutputStream());

                    System.out.println("New connection from " + socket.getInetAddress().getHostAddress());

                    String playerUUID = UUID.randomUUID().toString();

                    if (controllerUUID == null) {
                        controllerUUID = playerUUID;
                    }

                    lock.lock();
                    try {
                        connectedClient.put(playerUUID, socket);
                        clientOutput.put(playerUUID, output);
                    } finally {
                        lock.unlock();
                    }

                    JSONObject json = new JSONObject();
                    json.put("name","uuid");
                    json.put("value", playerUUID);
                    json.put("target", "player");
                    sendDataByUUID(json.toString(), playerUUID);

                    new Thread(() -> listen(socket, input, playerUUID)).start();

                } catch (IOException exception) {
                    if (!close)
                        System.out.println("Error while accepting connexion: " + exception.getMessage());
                } catch (JSONException e) {
                    throw new RuntimeException(e);
                }
            }
        }
        catch (IOException exception) {
            throw new RuntimeException("Error while starting the server: " + exception.getMessage(), exception);
        }
    }

    /**
     * Stop the server and close all existing connexion
     */
    public void stop() {
        close = true;
        try {
            if (serverSocket != null && !serverSocket.isClosed()) {
                serverSocket.close();
            }
            for (Socket socket : connectedClient.values()){
                socket.close();
            }
            System.out.println("Server is stopped");
        } catch (IOException exception) {
            System.err.println("Error while stopping the server: " + exception.getMessage());
        }
    }

    /**
     * Listen to the input stream and fetch data that are sent on it
     * Then process the data into the game controller
     *
     * @param input the input stream to be listened
     * @param socket the socket used for the connexion with the client that is sending data on the input
     */
    public void listen(Socket socket, ObjectInputStream input, String playerUUID) {
        while (!socket.isClosed()){
            try {
                String receivedObject = (String) input.readObject();
                JSONObject json = new JSONObject(receivedObject);


                if (json.getString("target").equals("player")){
                    //send json to the specified player
                    //most likely message from controller
                    sendDataByUUID(json.toString(), json.getString("uuid"));
                }
                else if (json.getString("target").equals("server")){
                    //send json to the controller
                    //most likely any player
                    sendDataByUUID(json.toString(), controllerUUID);
                } else if (json.getString("target").equals("all")) {
                    //send json to all client except the sender
                    //most likely the controller to update game state
                    sendDataAll(receivedObject, json.getString("uuid"));
                }
            }
            catch (ClassNotFoundException exception) {
                System.out.println("Error while receiving data: " + exception.getMessage());

            }
            catch (IOException exception){
                try {
                    System.out.println("IO error, socket is closing" + exception.getMessage());
                    socket.close();

                    lock.lock();
                    try {
                        connectedClient.remove(playerUUID);
                        clientOutput.remove(playerUUID);
                    } finally {
                        lock.unlock();
                    }

                    controllerUUID = null;
                }
                catch (IOException exception2){
                    System.out.println("Error while closing client connexion" + exception2.getMessage());
                }
            } catch (JSONException e) {
                throw new RuntimeException(e);
            }
        }

        lock.lock();
        try {
            connectedClient.remove(playerUUID);
            clientOutput.remove(playerUUID);
        } finally {
            lock.unlock();
        }
    }

    /**
     * Send the packet to every client currently connected except the one whose uuid is specified
     *
     * @param json the data to send
     */
    public void sendDataAll(String json, String playerUUID) {
        //clientOutput.removeIf(Objects::isNull);
        for (String clientUUID : clientOutput.keySet()){
            if (clientUUID.equals(playerUUID)) continue;

            ObjectOutputStream output = clientOutput.get(clientUUID);
            try {
                output.writeObject(json);
                output.flush();
            } catch (IOException exception) {
                System.out.println("I/O error while sending data to all: " + exception.getMessage());
            }
        }
    }

    /**
     * Send the packet to the client having the specified uuid
     *
     * @param playerUUID the id of the player to which the date must be send
     */
    public void sendDataByUUID(String json, String playerUUID){
        ObjectOutputStream output = clientOutput.get(playerUUID);
        try {
            output.writeObject(json);
            output.flush();
        } catch (IOException exception) {
            System.out.println("I/O error while sending data by uuid: " + exception.getMessage());
            throw new RuntimeException(exception);
        }
    }
}
